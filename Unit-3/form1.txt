    <script src="https://unpkg.com/react@17/umd/react.development.js" crossorigin></script>
    <script src="https://unpkg.com/react-dom@17/umd/react-dom.development.js" crossorigin></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
  </head>
  <body>
    <div id="root"></div>
    <script type="text/babel">
class Form extends React.Component{
constructor(props){
super(props);
this.state={
username:' ',
comments:' '
}
}

handlen=(event)=>{
this.setState({username:event.target.value})
}

handleC=(event)=>{
this.setState({comments:event.target.value})
}

handleS=(event)=>{
alert('A user name has entered'+this.state.username+this.state.comments);
}

render(){
return(
<form onSubmit={this.handleS}>
Username:
<input type="text" value={this.state.username} onChange={this.handlen}/>
Comment:
<textarea value={this.state.comments} onChange={this.handleC}/>
<input type="submit" value="submit"/>
</form>
)
}
}
ReactDOM.render(<Form />,document.getElementById("root"));
</script>
</body>
